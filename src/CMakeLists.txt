set(MODULE_NAME os)

if(${CMAKE_SYSTEM_NAME} STREQUAL "Windows")
    set(_PLATFORM_SOURCES win.cpp)
    set(_PLATFORM_DEFINITIONS SKLIB_PLATFORM_WINDOWS)
    elseif(${CMAKE_SYSTEM_NAME} STREQUAL "Darwin")
    set(_PLATFORM_SOURCES macos.cpp)
    set(_PLATFORM_DEFINITIONS SKLIB_PLATFORM_DARWIN)
    elseif(${CMAKE_SYSTEM_NAME} STREQUAL "Linux")
    set(_PLATFORM_SOURCES linux.cpp)
    set(_PLATFORM_DEFINITIONS SKLIB_PLATFORM_LINUX)
endif()

set(MODULE_SOURCE ${_PLATFORM_SOURCES})
set(MODULE_DEFINITIONS ${_PLATFORM_DEFINITIONS})
set(MODULE_TEST ON)
set(MODULE_NAME ${_SK_MODULE_NAME})

include(${CMAKE_SOURCE_DIR}/cmake/module.cmake)
